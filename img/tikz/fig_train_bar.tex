%% ── helper macro ─────────────────────────────────────────
%% \barrow{y-bot}{trainD}{valD}{testD}{total}
\newcommand{\barrow}[5]{
  \pgfmathsetmacro{\xTr}{#2*\sc}
  \pgfmathsetmacro{\xVa}{(#2+#3)*\sc}
  \pgfmathsetmacro{\xTe}{(#2+#3+#4)*\sc}
  % bars
  \fill[thesisBlue]    (0,  {#1}) rectangle (\xTr, {#1+\rh});
  \fill[thesisAmber]   (\xTr, {#1}) rectangle (\xVa, {#1+\rh});
  \fill[thesisCrimson] (\xVa, {#1}) rectangle (\xTe, {#1+\rh});
  % Train label inside
  \node[inlabel] at ({(\xTr)/2}, {#1+\rh/2}) {#2};
  % Val annotation: above bar (always narrow)
  \node[annlabel, above=2pt] at ({(\xTr+\xVa)/2}, {#1+\rh}) {#3};
  % Test label: inside if wide enough, else inside (force)
  \pgfmathsetmacro{\tw}{#4*\sc}
  \ifdim\tw cm > 0.8cm
    \node[inlabel] at ({(\xVa+\xTe)/2}, {#1+\rh/2}) {#4};
  \fi
  % Total label at right
  \node[totlabel] at ({\xTe+0.18}, {#1+\rh/2}) {Total #5};
}
