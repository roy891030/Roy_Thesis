%% ── x-scale: total = 2782 days → 10 cm, so 1cm = 278.2 days ─
\def\sc{0.003596}  % cm per day  (10/2782)
\def\rowh{0.85}    % bar height
\def\rowy{0.35}    % y offset in band

%% Day counts
% Short:  Train 414, Val 46, Test 116  (total 576)
% Medium: Train 1118,Val 124,Test 311  (total 1553)
% Long:   Train 2003,Val 222,Test 557  (total 2782)

\newcommand{\drawbarrow}[9]{%
  % #1=y-bot #2=Train #3=Val #4=Test #5=total
  % #6=train_lab #7=val_days #8=test_lab #9=total_label
  \pgfmathsetmacro{\xTrE}{#2*\sc}
  \pgfmathsetmacro{\xVaE}{(#2+#3)*\sc}
  \pgfmathsetmacro{\xTeE}{(#2+#3+#4)*\sc}
  % bars
  \fill[thesisBlue]    (0, {#1+\rowy}) rectangle (\xTrE, {#1+\rowy+\rowh});
  \fill[thesisAmber]   (\xTrE, {#1+\rowy}) rectangle (\xVaE, {#1+\rowy+\rowh});
  \fill[thesisCrimson] (\xVaE, {#1+\rowy}) rectangle (\xTeE, {#1+\rowy+\rowh});
  % train label inside bar
  \node[barlab] at ({(\xTrE)/2}, {#1+\rowy+\rowh/2}) {#6};
  % val annotation above (small, narrow bar)
  \node[annlab] at ({(\xTrE+\xVaE)/2}, {#1+\rowy+\rowh}) {#7};
  % test label inside bar (if wide enough)
  \pgfmathsetmacro{\testw}{#4*\sc}
  \ifdim\testw pt > 0.8cm
    \node[barlab] at ({(\xVaE+\xTeE)/2}, {#1+\rowy+\rowh/2}) {#8};
  \fi
  % total label right
  \node[totlab] at ({\xTeE+0.15}, {#1+\rowy+\rowh/2})
        {Total #5};
}

%% ── Draw rows ────────────────────────────────────────────────

\foreach \d in {0,500,1000,1500,2000,2500,3000}{

}
