%% ── helper macro ─────────────────────────────────────────
%% \barrow{y-bot}{trainD}{valD}{testD}{total}{pctTrain}{pctTest}
\newcommand{\barrow}[7]{
  \pgfmathsetmacro{\xTr}{#2*\sc}
  \pgfmathsetmacro{\xVa}{(#2+#3)*\sc}
  \pgfmathsetmacro{\xTe}{(#2+#3+#4)*\sc}
  % bars with rounded corners
  \fill[thesisBlue!85, rounded corners=2pt]    (0,  {#1}) rectangle (\xTr, {#1+\rh});
  \fill[thesisAmber!90, rounded corners=2pt]   (\xTr, {#1}) rectangle (\xVa, {#1+\rh});
  \fill[thesisCrimson!85, rounded corners=2pt] (\xVa, {#1}) rectangle (\xTe, {#1+\rh});
  % Thin white dividers
  \draw[white, line width=1.2pt] (\xTr, {#1}) -- (\xTr, {#1+\rh});
  \draw[white, line width=1.2pt] (\xVa, {#1}) -- (\xVa, {#1+\rh});
  % Outer border
  \draw[thesisDark!40, thick, rounded corners=2pt] (0, {#1}) rectangle (\xTe, {#1+\rh});
  % Train label inside
  \node[inlabel] at ({(\xTr)/2}, {#1+\rh/2}) {#2 d\quad(#6\%)};
  % Val annotation: above bar
  \node[annlabel, above=3pt] at ({(\xTr+\xVa)/2}, {#1+\rh}) {#3 d};
  % Test label: inside if wide enough
  \pgfmathsetmacro{\tw}{#4*\sc}
  \ifdim\tw cm > 0.8cm
    \node[inlabel] at ({(\xVa+\xTe)/2}, {#1+\rh/2}) {#4 d\quad(#7\%)};
  \fi
  % Total label at right
  \node[totlabel] at ({\xTe+0.22}, {#1+\rh/2}) {合計 #5 d};
}
