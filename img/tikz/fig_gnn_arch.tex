\begin{tikzpicture}[
  % ── generic node ─────────────────────────────────────────
  fnode/.style={circle, draw=#1!80!thesisDark, fill=#1!40, thick,
                minimum size=14mm, font=\normalsize\bfseries, inner sep=0},
  hnode/.style={circle, draw=thesisForest, fill=thesisForest!40, very thick,
                minimum size=14mm, font=\normalsize, inner sep=0},
  % ── arrows ───────────────────────────────────────────────
  arr/.style={-{Stealth[length=3mm,width=2.5mm]}, thesisDark!55, line width=1.0pt},
  laybox/.style={draw=thesisDark!40, dashed, rounded corners=4pt,
                 inner sep=10pt, line width=0.8pt},
  laylabel/.style={font=\small\itshape, thesisDark},
  eqlabel/.style={font=\normalsize, thesisDark, align=center},
]

%% ─── column positions & node y-spacing ──────────────────
\def\colsep{5.0}   % horizontal distance between column centres
\def\nodesep{2.2}  % vertical distance between nodes in a column
\def\N{5}          % nodes per column

% Colour cycle for input / output columns
\def\colorA{thesisTeal}
\def\colorB{thesisCrimson}
\def\colorC{thesisAmber}
\def\colorD{thesisPurple}
\def\colorE{thesisForest}

%% ─── Column 0: Input X ───────────────────────────────────
\foreach \row/\col in {1/\colorA, 2/\colorB, 3/\colorC, 4/\colorD, 5/\colorE}{
  \node[fnode=\col] (x\row) at (0, {(\N-\row)*\nodesep})
      {$x_{\row}$};
}
\begin{pgfonlayer}{background}
  \node[laybox, fit=(x1)(x5)] (box0) {};
\end{pgfonlayer}
\node[laylabel, below=4pt] at (box0.south) {input layer};
\node[eqlabel, below=16pt] at (box0.south) {$X$};

%% ─── Column 1: Hidden H^1 ───────────────────────────────
\foreach \row in {1,...,5}{
  \node[hnode] (h1\row) at (\colsep, {(\N-\row)*\nodesep}) {};
}
\begin{pgfonlayer}{background}
  \node[laybox, fit=(h11)(h15)] (box1) {};
\end{pgfonlayer}
\node[laylabel, below=4pt]  at (box1.south) {hidden layer};
\node[eqlabel, below=18pt] at (box1.south) {$H^1 = \delta(AXW^0)$};

%% ─── Column 2: Hidden H^2 ───────────────────────────────
\foreach \row in {1,...,5}{
  \node[hnode] (h2\row) at ({2*\colsep}, {(\N-\row)*\nodesep}) {};
}
\begin{pgfonlayer}{background}
  \node[laybox, fit=(h21)(h25)] (box2) {};
\end{pgfonlayer}
\node[laylabel, below=4pt]  at (box2.south) {hidden layer};
\node[eqlabel, below=18pt] at (box2.south) {$H^2 = \delta(AH^1W^1)$};

%% ─── Column 3: Output Z ─────────────────────────────────
\foreach \row/\col in {1/\colorB, 2/\colorE, 3/\colorC, 4/\colorA, 5/\colorD}{
  \node[fnode=\col] (z\row) at ({3*\colsep}, {(\N-\row)*\nodesep})
      {$z_{\row}$};
}
\begin{pgfonlayer}{background}
  \node[laybox, fit=(z1)(z5)] (box3) {};
\end{pgfonlayer}
\node[laylabel, below=4pt] at (box3.south) {output layer};
\node[eqlabel, below=16pt] at (box3.south) {$Z$};

%% ─── Arrows: X → H^1 (sampled, representative) ─────────
\foreach \fr/\to in {x1/h11, x1/h12, x2/h11, x3/h13, x4/h13,

  \draw[arr] (\fr.east) -- (\to.west);
}

%% ─── Arrows: H^1 → H^2 ──────────────────────────────────
\foreach \fr/\to in {h11/h21, h12/h21, h11/h22, h13/h22, h13/h23,

%% ─── Arrows: H^2 → Z ────────────────────────────────────
\foreach \fr/\to in {h21/z1, h21/z2, h22/z2, h23/z3, h23/z4,
                     h24/z4, h25/z5, h24/z1, h22/z3, h25/z5}{
  \draw[arr] (\fr.east) -- (\to.west);
