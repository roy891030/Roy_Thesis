%% ── Cluster macro ────────────────────────────────────────
%% Draws one 6-node fully-connected cluster (2 cols × 3 rows).
%% #1 = x offset,  #2 = node colour,  #3 = cluster title,
%% #4..#9 = six stock labels (top-left to bot-right, row-major)
%%
%% Node layout (local coords):
%%   ( -hg,  vg ) ( hg,  vg )   ← row 1 (top)
%%   ( -hg,   0 ) ( hg,   0 )   ← row 2 (mid)
%%   ( -hg, -vg ) ( hg, -vg )   ← row 3 (bot)

\newcommand{\cluster}[9]{%
  \def\hg{1.15}  % half horizontal gap (col offset)
  \def\vg{2.10}  % vertical gap (row offset) — enough room for label + edge
  \def\nr{0.55}  % node radius in cm (min size = 2*\nr = 11mm)
  %
  % Compute 6 node coordinates (in absolute coords)
  \coordinate (cA) at ({#1-\hg}, { \vg});
  \coordinate (cB) at ({#1+\hg}, { \vg});
  \coordinate (cC) at ({#1-\hg}, {  0 });
  \coordinate (cD) at ({#1+\hg}, {  0 });
  \coordinate (cE) at ({#1-\hg}, {-\vg});
  \coordinate (cF) at ({#1+\hg}, {-\vg});
  %
  % Draw ALL edges (complete graph, 15 pairs)
  \begin{pgfonlayer}{background}
    \foreach \ii/\jj in {cA/cB, cA/cC, cA/cD, cA/cE, cA/cF,
                         cB/cC, cB/cD, cB/cE, cB/cF,
                         cC/cD, cC/cE, cC/cF,
                         cD/cE, cD/cF, cE/cF}{
      \draw[arr] (\ii) -- (\jj);
    }
  \end{pgfonlayer}
  %
  % Draw nodes on top
  \foreach \nd in {cA,cB,cC,cD,cE,cF}{
    \node[circle, draw=#2!75!thesisDark, fill=#2!35, thick,
          minimum size={2*\nr cm}, inner sep=0]
         (N\nd) at (\nd) {};
    % Self-loop (outward)
    \draw[arr, looseness=7.5, out=100, in=80] (N\nd) to (N\nd);
  }
  %
  % Labels: above top nodes, beside middle nodes, below bottom nodes
  \node[nodefont, above=3pt] at (NcA.north) {#4};
  \node[nodefont, above=3pt] at (NcB.north) {#5};
  \node[nodefont, left=4pt]  at (NcC.west)  {#6};
  \node[nodefont, right=4pt] at (NcD.east)  {#7};
  \node[nodefont, below=3pt] at (NcE.south) {#8};
  \node[nodefont, below=3pt] at (NcF.south) {#9};
  %
  % Cluster title
  \node[font=\small\bfseries, thesisDark, anchor=south]
       at (#1, {\vg+0.9}) {#3};
}
