\begin{tikzpicture}[

]

\def\R{3.4}    % circle radius
\def\N{13}     % number of nodes

%% ── place nodes on circle ────────────────────────────────────
% 13 nodes, mix of colors to reflect multiple industries
\def\nodecolors{thesisTeal,thesisForest,thesisSand,thesisBlue,thesisTeal,thesisForest,
                thesisSand,thesisBlue,thesisTeal,thesisForest,thesisSand,thesisBlue,thesisTeal}
\def\nodelabels{2330\\台積電, 2317\\鴻海, 2454\\聯發科, 2882\\國泰金,
                3481\\群創, 2412\\中華電, 1301\\台塑, 2308\\台達電,
                2207\\和泰車, 5876\\上海商銀, 2395\\研華, 4904\\遠傳,
                2886\\兆豐金}

\foreach \i in {1,...,\N}{
  \pgfmathsetmacro{\angle}{90 + 360/\N * (\i-1)}
  \pgfmathsetmacro{\xp}{\R * cos(\angle)}
  \pgfmathsetmacro{\yp}{\R * sin(\angle)}
  % cycle through colors
  \pgfmathtruncatemacro{\ci}{mod(\i-1,4)}
  \ifnum\ci=0 \def\ncol{thesisTeal}   \fi
  \ifnum\ci=1 \def\ncol{thesisForest} \fi
  \ifnum\ci=2 \def\ncol{thesisSand}   \fi
  \ifnum\ci=3 \def\ncol{thesisBlue}   \fi
  \node[gnode=\ncol] (m\i) at (\xp, \yp) {};
  % self-loop
  \pgfmathsetmacro{\outang}{\angle+15}
  \pgfmathsetmacro{\inang}{\angle-15}
  \draw[selfarr, looseness=6, out=\outang, in=\inang] (m\i) to (m\i);
}

%% ── draw all edges (complete graph) ─────────────────────────
\begin{pgfonlayer}{background}
  \foreach \i in {1,...,\N}{
    \foreach \j in {1,...,\N}{
      \ifnum\i<\j
        \draw[edgestyle] (m\i) -- (m\j);
      \fi
    }
  }
\end{pgfonlayer}

%% ── center label ─────────────────────────────────────────────
