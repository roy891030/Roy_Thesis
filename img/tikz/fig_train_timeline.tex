%% ── helper: draw one Gantt row ───────────────────────────
\newcommand{\ganttrow}[8]{
  % Train bar (with rounded right end via slight overlap trick)
  \fill[thesisBlue!85, rounded corners=2pt]    (#2, {#1})   rectangle (#3, {#1+\rh});
  % Val bar
  \fill[thesisAmber!90, rounded corners=2pt]   (#3, {#1})   rectangle (#4, {#1+\rh});
  % Test bar
  \fill[thesisCrimson!85, rounded corners=2pt] (#4, {#1})   rectangle (#5, {#1+\rh});
  % Thin dividers between segments
  \draw[white, line width=1.2pt] (#3, {#1}) -- (#3, {#1+\rh});
  \draw[white, line width=1.2pt] (#4, {#1}) -- (#4, {#1+\rh});
  % Outer border
  \draw[thesisDark!40, thick, rounded corners=2pt] (#2, {#1}) rectangle (#5, {#1+\rh});
  % Train label (inside bar)
  \node[inlabel] at ({(#2+#3)/2}, {#1+\rh/2}) {#6};
  % Val annotation (above bar, since narrow)
  \node[annlabel, above=3pt] at ({(#3+#4)/2}, {#1+\rh}) {#7};
  % Test label
  \pgfmathsetmacro{\testwidth}{(#5-#4)}
  \ifdim\testwidth cm > 1.2cm
    \node[inlabel] at ({(#4+#5)/2}, {#1+\rh/2}) {#8};
  \else
    \node[annlabel, right=3pt] at (#5, {#1+\rh/2}) {#8};
  \fi
}
