%% ── helper: draw one bar row ────────────────────────────────
%% #1=y-bottom, #2=x_start, #3=x_trainEnd, #4=x_valEnd, #5=x_end
%% #6=Train label, #7=Val label, #8=Test label
\newcommand{\drawrow}[8]{%
  % Train bar
  \fill[thesisBlue] ({#2*\sc},{#1+\rowy}) rectangle ({#3*\sc},{#1+\rowy+\rowh});
  % Val bar
  \fill[thesisAmber] ({#3*\sc},{#1+\rowy}) rectangle ({#4*\sc},{#1+\rowy+\rowh});
  % Test bar
  \fill[thesisCrimson] ({#4*\sc},{#1+\rowy}) rectangle ({#5*\sc},{#1+\rowy+\rowh});
  % Train label (only if wide enough)
  \node[barlab] at ({(\sc*(#2+#3)/2)}, {#1+\rowy+\rowh/2}) {#6};
  % Val annotation (above bar, since too narrow)
  \node[annlab, above=2pt] at ({(\sc*(#3+#4)/2)}, {#1+\rowy+\rowh}) {#7};
  % Test label
  \node[barlab] at ({(\sc*(#4+#5)/2)}, {#1+\rowy+\rowh/2}) {#8};
}
